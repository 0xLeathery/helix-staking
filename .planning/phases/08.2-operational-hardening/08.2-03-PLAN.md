### Task 1: Fix indexer signature pagination (OPS-04 — HIGH)

**File:** `services/indexer/src/worker/poller.ts`

Replace the single `getSignaturesForAddress(programId, { limit: 1000, until: lastSig })` call with a pagination loop:

```typescript
async function fetchAllNewSignatures(
  connection: Connection,
  programId: PublicKey,
  lastSignature: string | null
): Promise<ConfirmedSignatureInfo[]> {
  const allSignatures: ConfirmedSignatureInfo[] = [];
  let before: string | undefined = undefined;

  while (true) {
    const batch = await connection.getSignaturesForAddress(programId, {
      limit: 1000,
      until: lastSignature || undefined,
      before,
    });

    if (batch.length === 0) break;
    allSignatures.push(...batch);

    // If we got fewer than 1000, we've reached the end
    if (batch.length < 1000) break;

    // Use the oldest signature in this batch as the cursor for the next page
    before = batch[batch.length - 1].signature;

    logger.info(`Fetched ${batch.length} signatures, paginating...`);
  }

  // Reverse to oldest-first for chronological processing
  return allSignatures.reverse();
}
```

```typescript
// Simple sliding window per IP
const requestCounts = new Map<string, { count: number; resetAt: number }>();
const RATE_LIMIT = 120; // per minute
const WINDOW_MS = 60_000;

function checkRateLimit(ip: string): boolean {
  const now = Date.now();
  const entry = requestCounts.get(ip);
  if (!entry || now > entry.resetAt) {
    requestCounts.set(ip, { count: 1, resetAt: now + WINDOW_MS });
    return true;
  }
  entry.count++;
  return entry.count <= RATE_LIMIT;
}
```

Modify the RPC client factory to attempt failover on exhausted retries:

```typescript
export function createRpcClient(): Connection {
  const primaryConnection = new Connection(env.RPC_URL, "confirmed");
  return primaryConnection;
}

export function createFallbackRpcClient(): Connection | null {
  if (!env.RPC_URL_FALLBACK) return null;
  return new Connection(env.RPC_URL_FALLBACK, "confirmed");
}

// In retry logic: after primary retries exhausted, switch to fallback
export async function withRpcFallback<T>(
  fn: (connection: Connection) => Promise<T>
): Promise<T> {
  try {
    return await pRetry(() => fn(createRpcClient()), { retries: 5 });
  } catch (primaryErr) {
    const fallback = createFallbackRpcClient();
    if (!fallback) throw primaryErr;
    logger.warn("Primary RPC failed, attempting fallback...");
    return await pRetry(() => fn(fallback), { retries: 3 });
  }
}
```
