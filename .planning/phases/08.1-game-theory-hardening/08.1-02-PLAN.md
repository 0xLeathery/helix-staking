<objective>
Add a duration loyalty multiplier that rewards stakers proportionally for time already served. Stakes that are closer to maturity earn higher daily inflation rewards than fresh stakes with identical T-shares.

Purpose: Currently there is no on-chain incentive to remain staked beyond the penalty deterrent. A loyalty multiplier creates a positive incentive that increases over time, strengthening the hold mechanic.
Output: claim_rewards and unstake apply a loyalty bonus of 0% → 50% linear over the stake's committed term.
</objective>

    ```rust
    pub const LOYALTY_MAX_BONUS: u64 = 500_000_000;
    ```

    **In `math.rs`:**
    ```rust
    /// Calculate loyalty bonus based on proportion of committed term already served.
    /// Returns bonus in PRECISION units (0 to LOYALTY_MAX_BONUS).
    ///
    /// Formula: loyalty_bonus = min(days_served / committed_days, 1) × LOYALTY_MAX_BONUS
    ///
    /// Examples (with LOYALTY_MAX_BONUS = 0.5x):
    /// - Day 0 of 365-day stake: 0% bonus
    /// - Day 182 of 365-day stake: ~25% bonus
    /// - Day 365 of 365-day stake: 50% bonus (max)
    /// - Day 400 of 365-day stake (in grace period): 50% bonus (capped)
    pub fn calculate_loyalty_bonus(
        start_slot: u64,
        current_slot: u64,
        committed_days: u64,
        slots_per_day: u64,
    ) -> Result<u64> {
        if committed_days == 0 || slots_per_day == 0 {
            return Ok(0);
        }

        let elapsed_slots = current_slot.saturating_sub(start_slot);
        let days_served = elapsed_slots.checked_div(slots_per_day).unwrap_or(0);

        // Cap at committed_days (don't exceed max bonus even in grace/late period)
        let capped_days = days_served.min(committed_days);

        // loyalty_bonus = capped_days * LOYALTY_MAX_BONUS / committed_days
        mul_div(capped_days, LOYALTY_MAX_BONUS, committed_days)
    }
    ```

    Add unit tests for the new function:
    ```rust
    #[test]
    fn test_loyalty_bonus() {
        // Day 0: no bonus
        assert_eq!(calculate_loyalty_bonus(0, 0, 365, 216_000).unwrap(), 0);

        // Half-term: half of max bonus (250_000_000 = 25%)
        let half = calculate_loyalty_bonus(0, 182 * 216_000, 365, 216_000).unwrap();
        assert!(half > 240_000_000 && half < 260_000_000); // ~25% ± rounding

        // Full term: max bonus (500_000_000 = 50%)
        let full = calculate_loyalty_bonus(0, 365 * 216_000, 365, 216_000).unwrap();
        assert_eq!(full, LOYALTY_MAX_BONUS);

        // Past term (grace period): still capped at max
        let over = calculate_loyalty_bonus(0, 400 * 216_000, 365, 216_000).unwrap();
        assert_eq!(over, LOYALTY_MAX_BONUS);

        // 1-day stake, day 1: max bonus
        let min_stake = calculate_loyalty_bonus(0, 216_000, 1, 216_000).unwrap();
        assert_eq!(min_stake, LOYALTY_MAX_BONUS);
    }
    ```

    ```typescript
    export const LOYALTY_MAX_BONUS = new BN('500000000'); // 0.5x in PRECISION

    /**
     * Calculate loyalty bonus based on proportion of committed term served.
     * Returns bonus in PRECISION units (0 to LOYALTY_MAX_BONUS).
     */
    export function calculateLoyaltyBonus(
      startSlot: BN,
      currentSlot: BN,
      committedDays: BN,
      slotsPerDay: BN,
    ): BN {
      if (committedDays.isZero() || slotsPerDay.isZero()) return new BN(0);

      const elapsedSlots = BN.max(currentSlot.sub(startSlot), new BN(0));
      const daysServed = elapsedSlots.div(slotsPerDay);
      const cappedDays = BN.min(daysServed, committedDays);

      return mulDiv(cappedDays, LOYALTY_MAX_BONUS, committedDays);
    }

    /**
     * Apply loyalty multiplier to pending rewards.
     * Returns: rewards × (1 + loyaltyBonus / PRECISION)
     */
    export function applyLoyaltyMultiplier(pendingRewards: BN, loyaltyBonus: BN): BN {
      const totalMultiplier = PRECISION.add(loyaltyBonus);
      return mulDiv(pendingRewards, totalMultiplier, PRECISION);
    }
    ```
