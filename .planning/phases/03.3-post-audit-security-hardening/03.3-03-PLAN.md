    }

    ///   - bpd_finalize_period_id: 0 (never finalized)
    pub fn migrate_stake(_ctx: Context<MigrateStake>) -> Result<()> {

        // - Zero-initializes new bytes
        // - Transfers rent from payer
        // If account is already at LEN, this is a no-op.
