pub const STAKE_SEED: &[u8] = b"stake";

pub const MAX_STAKE_DAYS: u64 = 5555;     // Maximum stake duration
pub const LPB_MAX_DAYS: u64 = 3641;       // 10 years for full 2x LPB bonus

// Penalty parameters
pub const MIN_PENALTY_BPS: u64 = 5000;     // 50% minimum early unstake penalty
pub const BPS_SCALER: u64 = 10_000;
pub const GRACE_PERIOD_DAYS: u64 = 14;
pub const LATE_PENALTY_BPS_PER_DAY: u64 = 29; // ~1% per ~3.5 days → 100% at 350 days past grace
// Total late days to 100% = GRACE_PERIOD_DAYS + (BPS_SCALER / LATE_PENALTY_BPS_PER_DAY) ≈ 14 + 345 ≈ 359
// Per roadmap: "linear to 100% over 350 days, total loss after 365"
// So 365 - 14 = 351 penalty days → BPS_SCALER / 351 ≈ 28.5, round up to 29

**instructions/math.rs** -- Create new file with all math functions. Use `checked_mul`, `checked_add`, `checked_sub`, `checked_div` for every arithmetic operation. Return `HelixError::Overflow` on failure.

Implement these public functions:

1. **`calculate_lpb_bonus(stake_days: u64) -> Result<u64>`**
   - Returns bonus multiplier scaled by PRECISION
   - Formula: `(days - 1) * 2 * PRECISION / LPB_MAX_DAYS`
   - Cap at `2 * PRECISION` for days >= LPB_MAX_DAYS
   - 1 day returns 0 bonus
   - Uses LPB_MAX_DAYS (3641) not MAX_STAKE_DAYS (5555)

2. **`calculate_bpb_bonus(staked_amount: u64) -> Result<u64>`**
   - Returns bonus multiplier scaled by PRECISION
   - Formula: `min(amount, BPB_THRESHOLD) * PRECISION / (10 * BPB_THRESHOLD)`
   - Cap at `PRECISION / 10` (0.1x = 10%)

3. **`calculate_t_shares(staked_amount: u64, stake_days: u64, share_rate: u64) -> Result<u64>`**
   - Combines LPB + BPB: `total_multiplier = PRECISION + lpb + bpb`
   - `effective_amount = staked_amount * total_multiplier / PRECISION`
   - `t_shares = effective_amount * PRECISION / share_rate`
   - Require share_rate > 0

4. **`calculate_early_penalty(staked_amount: u64, start_slot: u64, current_slot: u64, end_slot: u64) -> Result<u64>`**
   - If current_slot >= end_slot: return 0 (not early)
   - `served_fraction_bps = elapsed_slots * BPS_SCALER / total_slots`
   - `penalty_bps = BPS_SCALER - served_fraction_bps`
   - Enforce minimum: `max(penalty_bps, MIN_PENALTY_BPS)`
   - Return `staked_amount * final_penalty_bps / BPS_SCALER`

5. **`calculate_late_penalty(staked_amount: u64, end_slot: u64, current_slot: u64, slots_per_day: u64) -> Result<u64>`**
   - If current_slot <= end_slot: return 0 (not late)
   - `late_days = (current_slot - end_slot) / slots_per_day`
   - If late_days <= GRACE_PERIOD_DAYS: return 0
   - `penalty_days = late_days - GRACE_PERIOD_DAYS`
   - `penalty_bps = penalty_days * LATE_PENALTY_BPS_PER_DAY`
   - Cap at BPS_SCALER (100%)
   - Return `staked_amount * capped_penalty_bps / BPS_SCALER`

6. **`calculate_pending_rewards(t_shares: u64, current_share_rate: u64, reward_debt: u64) -> Result<u64>`**
   - `current_value = t_shares * current_share_rate / PRECISION`
   - `debt_value = reward_debt / PRECISION`
   - Return `current_value - debt_value` (saturating to 0 if negative due to rounding)

7. **`get_current_day(init_slot: u64, current_slot: u64, slots_per_day: u64) -> Result<u64>`**
   - `elapsed = current_slot - init_slot`
   - Return `elapsed / slots_per_day`
